مراجعة عملي – أساسيات قواعد البيانات

خطوة 1: إنشاء قاعدة البيانات

إنشاء قاعدة بيانات باسم Classes


CREATE DATABASE IF NOT EXISTS Classes;
USE Classes;
تم إنشاء قاعدة البيانات Classes وتحديدها للاستخدام

جدول 1: Teachers

بيانات المعلمين مع فهرس على اسم المعلم


CREATE TABLE Teachers (
    TeacherID INT AUTO_INCREMENT PRIMARY KEY,
    Teacher_name VARCHAR(40),
    CourseID INT,
    Salary INT
);

CREATE INDEX idx_teacher ON Teachers (Teacher_name);

INSERT INTO Teachers (Teacher_name, CourseID, Salary) VALUES
('Sara', 1, 5000),
('Rawan', 2, 6200),
('Maryam', 1, 5500),
('Reem', 3, 7000);
تم إنشاء جدول Teachers مع الفهرس وإدخال 4 سجلات

جدول 2: Courses

بيانات المقررات الدراسية


CREATE TABLE Courses (
    CourseID INT PRIMARY KEY,
    Course_name VARCHAR(20)
);

INSERT INTO Courses (CourseID, Course_name) VALUES
(1, 'Computer'),
(2, 'Math'),
(3, 'English');
تم إنشاء جدول Courses وإدخال 3 سجلات

جدول 3: Students

بيانات الطلاب المرتبطين بالمقررات


CREATE TABLE Students (
    StudentID INT AUTO_INCREMENT PRIMARY KEY,
    Student_name VARCHAR(40),
    CourseID INT,
    EnrollmentDate DATE,
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID)
);

INSERT INTO Students (Student_name, CourseID, EnrollmentDate) VALUES
('Lena', 1, '2025-09-01'),
('Noor', 2, '2025-09-02'),
('Dalal', 3, '2025-09-03');
تم إنشاء جدول Students وإدخال 3 سجلات

استعلام: عرض بيانات المعلمات براتب أكثر من 5500


SELECT * 
FROM Teachers
WHERE Salary > 5500
ORDER BY Salary DESC;
